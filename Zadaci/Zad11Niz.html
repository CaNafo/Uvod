/*Ispitati da li je uneseni niz palindrom.*/

#include <stdio.h>
main()

{

	char niz[30], niz1[30];
	int i, d = 0, j = 0;

	scanf("%s", &niz);

	for (i = 0; niz[i] != '\0'; i++)

		d++;
	for (i = d - 1; i >= 0; i--)

		niz1[j++] = niz[i];

	niz1[j] = '\0';
	if (strcmp(niz, niz1) == 0)

		printf("Uneseni niz je palindrom!\n");

	else
		printf("Uneseni niz nije palindrom!\n");

	system("PAUSE");

}

/*Omogućiti unošenje stringova sve dok se ne unese palindrom. Kada se unese palindrom ispisati poruku „Unesen je palindrom!“.*/

#include <stdio.h>

main()

{
	char niz[30], niz1[30];

	int i, d = 0, j = 0;

	do {
		scanf("%s", &niz);

		for (i = 0; niz[i] != '\0'; i++)

			d++;
		for (i = d - 1; i >= 0; i--)

			niz1[j++] = niz[i];

		niz1[j] = '\0';

		if (strcmp(niz, niz1) == 0)
		{

			printf("Uneseni niz je palindrom!\n");

		}
		j = 0;

		d = 0;

	} while (strcmp(niz, niz1) != 0);

	system("PAUSE");

}

/*Napisati program koji pronalazi broj pojavljivanja trazenog pojma u unesenom tekstu.*/

#include <stdio.h>

main()
{

	int i, j, duz_tekst, duz_trazeni_pojam, p = 0;

	char tekst[64], trazeni_pojam[64], pom[64];

	printf("Unesite tekst: ");

	gets(tekst);

	printf("Unesite trazeni pojam/tekst za pretragu: "); gets(trazeni_pojam);

	duz_tekst = strlen(tekst);

	duz_trazeni_pojam = strlen(trazeni_pojam);
	for (i = 0; i < duz_tekst; i++)

	{

		if (tekst[i] == trazeni_pojam[0])
		{

			for (j = i; j < duz_trazeni_pojam + i; j++)

			{

				pom[j - i] = tekst[j];

			}

			pom[j - i] = '\0';
			if (strcmp(pom, trazeni_pojam) == 0)

				p++;

		}

	}

	printf("Pronadjeno je ukupno: %d\n ", p);
	printf("\n\n\n");

	system("PAUSE");

}
