/*Napisati program na C-u u koji se unose granice A i B tako da je A < B ukoliko uslov nije ispunjen tražiti ponovni unos granica. Zatim unijeti broj N i na ekranu ispisati sve brojeve iz skupa [A,B] djeljive sa brojem N.*/

#include <stdio.h>
main()
{
	int A, B, N, brojac;
	
	do {
		printf("Unesite granice A i B: ");
		scanf("%d %d", &A, &B);
	} while ((A > B) || (A==B));
	
	printf("Unesite broj N: ");
	scanf("%d", &N);

	brojac = A;
	
	do {
		if (brojac%N == 0)
			printf("Broj %d je djeljiv brojem %d\n", brojac, N);
			brojac++;
	} while (brojac <= B);

	system("pause");
}

/*Napisati program na C-u u koji se unose po tri broja sve dok su prvi i treći broj jednaki. Na ekran štampati proizvod svih drugih parnih brojeva. Ako je ovaj proizvod jednak jedinici na ekran štampati: "Proizvod svih drugih parnih brojeva je jedan".*/

#include<stdio.h>
main()
{
	int a, b, c, proizvod = 1;
	do
	{
		printf("Unesi brojeve a,b,c: ");
		scanf_s("%d %d %d", &a, &b, &c);
		if (a == c)
		{
			if (b % 2 == 0)
				proizvod *= b;
		}
	} while (a == c);
	if (proizvod == 1)
		printf("\nProizvod svih drugih parnih brojeva je jedan.\n");
	else
		printf("Proizvod je %d\n", proizvod);
	system("pause");
}

/*Napisati program na C-u u koji se unose parovi cijelih brojeva sve dok je jedan od brojeva paran a drugi neparan(ne zna se koji je koji). Štampati proizvod parnih i aritmetičku sredinu neparnih brojeva ,bez para koji je prekinuo unos*/

#include<stdio.h>
main()
{
	int a, b, proizvod = 1, suma = 0, brojac = 0;
	do
	{
		printf("Unesi brojeve: ");
		scanf_s("%d %d", &a, &b);
		if ((a % 2 == 0 && b % 2 != 0) || (a % 2 != 0 && b % 2 == 0))
		{
			if (a % 2 == 0)
			{
				proizvod *= a;
				suma += b;
				brojac++;
			}
			else
			{
				suma += a;
				proizvod *= b;
				brojac++;
			}
		}
	} while ((a % 2 == 0 && b % 2 != 0) || (a % 2 != 0 && b % 2 == 0));
	printf("Proizvod je %d, aritmeticka sredina je %.2f\n", proizvod, (float)suma / brojac);
	system("pause");
}
