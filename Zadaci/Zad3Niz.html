/*Unijeti niz karaktera sa tastature, a zatim napraviti novi niz karaktera koji će biti identičan prvoj polovini unesenog niza karaktera.*/

#include <stdio.h>

main()

{

	char niz[20], niz1[10];

	int i, duzina = 0;
	printf("Unesite niz karaktera: ");
	scanf("%s", &niz);

	for (i = 0; niz[i] != '\0'; i++)

		duzina++;
	for (i = 0; i<duzina / 2; i++)

		niz1[i] = niz[i];

	niz1[duzina / 2] = '\0';
	printf("\nNovi niz je: %s\n", niz1);

	system("PAUSE");

}

/*Unijeti dva niza karaktera sa tastature, a zatim napraviti novi niz karaktera u koji treba nadovezati drugi na prvi uneseni niz karaktera.*/

#include <stdio.h>

main()

{

	char niz[20], niz1[10], niz2[10];
	printf("Unesite dva niza karaktera: ");
	int i, d = 0;
	scanf("%s%s", &niz1, &niz2);

	for (i = 0; niz1[i] != '\0'; i++)

		niz[d++] = niz1[i];
	for (i = 0; niz2[i] != '\0'; i++)

		niz[d++] = niz2[i];

	niz[d] = '\0';

	printf("\nNovi niz je: %s\n", niz);
	system("PAUSE");

}
