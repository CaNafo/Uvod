/*Napisati program na programskom jeziku C koji omogucava da korisnik unese 'n' brojeva, a zatim se na ekranu ispisuje najveci i najmanji od brojeva*/

#include<stdio.h>
main()
{
	int n,i,broj;
	int min, max;

	printf("Koliko brojeva zelite unijeti? ");
	scanf("%d", &n);

	printf("Unesite 1. broj: ");
	scanf("%d", &broj);
	min = broj;
	max = broj;

	for (i = 1; i < n; i++)
	{
		printf("Unesite %d. broj: ", i + 1);
		scanf("%d", &broj);
		if (broj > max)
			max = broj;
		if (broj < min)
			min = broj;
	}

	printf("Najveci uneseni broj je %d, a najmanji uneseni broj je %d\n", max, min);

	system("pause");
}

/*Napisati program u programskom jeziku C koji ispisuje koliko je brojeva od 10 do 100 koji su djeljivi sa 7 ili sa 5.*/

#include <stdio.h>

 main()
{
	int i, br = 0;

	for (i = 10; i <= 100; i++)
		if (i % 7 == 0 || i % 5 == 0)
			br++;
	printf("\nBrojeva djeljivih sa 7 ili 5 ima %d.\n", br);

	system("pause");
}
