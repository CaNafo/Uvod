/*Ucitati niz karaktera sa tastature, a potom formirati novi niz karaktera na osnovu ucitanog niza tako što će se svako pojavljivanje neke cifre zamijeniti sa „CC“*/

#include <stdio.h>

main()

{
	char niz[20], niz1[40];

	int i, j = 0;
	printf("Unesite string: ");
	scanf("%s", &niz);
	for (i = 0; niz[i] != '\0'; i++)

		if (niz[i] >= '0'&&niz[i] <= '9')

		{
			niz1[j] = 'C';

			niz1[j + 1] = 'C';

			j += 2;
		}

		else

		{

			niz1[j] = niz[i];
			j++;

		}

	niz1[j] = '\0';
	printf("Trazeni niz je: %s\n", niz1);

	system("PAUSE");

}

/*Ucitati niz karaktera sa tastature. Odštampati taj niz ali tako da mala slova pretvori u velika.*/

#include <stdio.h>
main()

{

	char niz[30];
	int i, j = 0;
	printf("Unesite niz karaktera: ");
	scanf("%s", &niz);

	for (i = 0; niz[i] != '\0'; i++)

		if (niz[i] >= 'a'&&niz[i] <= 'z')
			niz[i] += 'A' - 'a';

	printf("Trazeni niz je: %s\n", niz);

	system("PAUSE");

}
