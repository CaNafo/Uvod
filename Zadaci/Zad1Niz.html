/*Napisati program na programskom jeziku C u koji se unosi i ispisuje niz cijelih brojeva.*/

#include<stdio.h>
main()
{
	int niz[100], n, i;
	
	printf("Koliko brojeva zelite unijeti u niz: ");
	scanf("%d", &n);

	for (i = 0; i < n; i++)
	{
		printf("Unesite %d. broj: ", i + 1);
		scanf("%d", &niz[i]);
	}

	printf("\nUneseni niz je: \n");

	for (i = 0; i < n; i++)
	{
		printf("%d ", niz[i]);
	}
	printf("\n");
	system("pause");
}

/*Napisati program na progamskom jeziku C u koji se unosi niz cijelih brojeva, zatim se prebrojava i ispisuje koliko ima parnih brojeva u nizu. Ispisati te brojeve.*/

#include<stdio.h>
main()
{
	int niz[100], n, i, brojParnih = 0;
	
	printf("Koliko brojeva zelite unijeti u niz: ");
	scanf("%d", &n);

	for (i = 0; i < n; i++)
	{
		printf("Unesite %d. broj: ", i + 1);
		scanf("%d", &niz[i]);
		if (niz[i] % 2 == 0)
			brojParnih++;
	}

	printf("\Parnih brojeva ima %d i to su: \n",brojParnih);

	for (i = 0; i < n; i++)
	{
		if (niz[i] % 2 == 0)
			printf("%d ", niz[i]);
	}
	printf("\n");
	system("pause");
}
