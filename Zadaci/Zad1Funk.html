/*Napisati funkciju "Sabiranje" kojoj se proslijeđuju sabirci, a na ekran ispisuje njihov zbir*/

#include<stdio.h>
#include<stdlib.h>
void Sabiranje(int a, int b); //Navodjenje potpisa funkcije (Moguce je napisati potpunu funkicju iznad main metode
							//ali zbog preglednosti se praktikuje njeno definisanje ispod main metode)
int main()
{
	int a, b;
	printf("Unesite sabirke a i b: ");
	scanf("%d %d", &a, &b);

	Sabiranje(a, b); //poziv funkcije

	system("pause");

}
void Sabiranje(int a, int b) {  //Funkcija je tipa VOID sto znaci da ne vraca nista, a da bi mogla funkcionisati potrebno joj je proslijediti
								// u ovom slucaju dva paramtetra tipa int
	int zbir;
	
	zbir = a + b;

	printf("Zbir %d + %d = %d\n", a, b, zbir);
}

/*Napisati funkciju "Sabiranje" kojoj se proslijeđuju sabirci, a funkcija vraća vrijednost sabiranja i u main metodi se vrši ispis rezultata*/

#include<stdio.h>
#include<stdlib.h>
int Sabiranje(int a, int b); //Promjenjive koje se proslijedjuju i promjenjive koje su navedene kao parametri funkcije ne moraju imati isto ime
							//u funkciji se radi samo sa vrijednostima promjenjivih, te nakon zavrsetka rada funkcije sve promjenjive unutar funkcije se uništavaju.
int main()
{
	int a, b, zbir;
	printf("Unesite sabirke a i b: ");
	scanf("%d %d", &a, &b);

	zbir = Sabiranje(a, b); //poziv funkcije promjenjivj zbir se dodjeljuje vrijednost koja je izracunata unutar funkcije

	printf("Zbir %d + %d = %d \n", a, b, zbir);

	//printf("Zbir %d + %d = %d \n", a, b, Sabiranje(a, b));  Moguce je i ovako napisati, tj. bez pomocne promjenjive "zbir"
	
	system("pause");

}
int Sabiranje(int a, int b) {  //Funkcija je tipa int sto znaci da vraca podatak tipa int, a da bi mogla funkcionisati potrebno joj je proslijediti
								// u ovom slucaju dva paramtetra tipa int
	int zbir;
	
	zbir = a + b;

	return zbir; //Funkcija vraca podatak tipa int
 // return a+b;  moguce je i ovako napisati
}
