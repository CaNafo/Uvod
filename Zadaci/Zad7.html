/*Napisati program u programskom jeziku C u kome se unosi cijeli broj, a zatim na ekranu ispisuje da li je broj paran ili neparan */

#include <stdio.h>

main()
{
    int broj;

    printf("\nUnesite cijeli broj: "); 
    scanf("%d", &broj);

    if(broj % 2 != 0) 
        printf("\nBroj je neparan"); 
    else 
        printf("\nBroj je paran"); 

    system("pause");
}

/*Napisati program u programskom jeziku C u kome je potrebno sa tastature unijeti neki karakter, a zatim na ekranu ispisati poruku da li je taj karakter slovo, broj ili neki znak */

#include <stdio.h>

main()
{
  	char ch;
  	
  	printf("Unesite neki karakter: ");
  	scanf("%c",&ch);
  	
  	if((ch>='a' && ch<='z') || (ch>='A' && ch<='Z'))
  	{
  	 printf("Unešeni karakter je slovo!");
  	}
          else if(ch>='0' && ch<='9')
          {
            printf("Unešeni karakter je broj!");
          }
        else 
       {
            printf("Unešeni karakter je znak!");
       }   
    
    system("pause");
}
