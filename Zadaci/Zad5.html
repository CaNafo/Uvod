/*Napisati program u C-u kojim se omogućava unos broja dana. Izračunati koliko to iznosi godina, mjeseci i dana*/

#include <stdio.h>

 main()
{
    int ukupno, god, mjes, dana;

    printf("\nUnesite broj dana: ");
    scanf("%d", &ukupno);

    god  = ukupno / 365;
    mjes = (ukupno % 365) / 30;
    dana = (ukupno % 365) % 30;

    printf("\n%d dani = %d god, %d mjes i %d dana", ukupno, god, mjes, dana); 

    system("pause");
}
