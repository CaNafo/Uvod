Unos i ispis niza 2

/*Napisati program na progamskom jeziku C u koji se unosi niz cijelih brojeva, zatim se svaki neparan broj zamjenjuje nulom. Ispisati niz prije i poslije transformacije.*/

#include<stdio.h>
main()
{
	int niz[100], n, i;
	
	printf("Koliko brojeva zelite unijeti u niz: ");
	scanf("%d", &n);

	for (i = 0; i < n; i++)
	{
		printf("Unesite %d. broj: ", i + 1);
		scanf("%d", &niz[i]);
	}

	printf("\Niz prije transformacije je: \n");
	for (i = 0; i < n; i++)
	{
		printf("%d ", niz[i]);
	}
	for (i = 0; i < n; i++)
	{
		if (niz[i] % 2 != 0)
			niz[i] = 0;
	}
	printf("\n");
	printf("\Niz poslije transformacije je: \n");
	for (i = 0; i < n; i++)
	{
		printf("%d ", niz[i]);
	}
	printf("\n");
	system("pause");
}

/* Unijeti niz karaktera sa tastature. Na ekran ispisati dužinu tog niza.*/

	char niz[20];

Napomena!!!
U programskom jeziku C nizovi karaktera se ponašaju drugačije od nizova brojeva. Na kraju svakog niza karaktera se automatski dodaje karkater '\0'. Taj karakter označava kraj niza. Na taj način omogućeno je koršćenje nizova karaktera, a da unaprijed ne znamo dužinu tog niza. Kada hoćemo da prođemo kroz čitav niz, krećemo se sve dok ne dođemo do '\0'. Nizovi karaktera se mogu unijeti i štampati u cjelini, a ne element po element kao što je u slučaju sa nizovima brojeva.
Koja je razlika izmedju „a“ i 'a'?
'a' je karakter , a „a“ je niz karaktera koji se sastoji od 'a' i '\0'.
