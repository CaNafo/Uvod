/*Napisati program na programskom jeziku C omogućava učitavanje pozitivnih brojeva sa tastautre, a zatim kada se pojavi prvi negativni broj na ekranu ispisati zbir do tada unesenih brojeva (Računati i broj 0 u pozitvne brojeve)*/

#include <stdio.h>
 main()
{
	 int suma = 0,broj,i=2;

	 printf("Unesite 1 broj: ");
	 scanf("%d", &broj);

	 if (broj >= 0)  //Ispitivanje da li je prvi broj pozitivan ili nula
		 suma = suma + broj;
	 else
		 printf("Nema ni jedan pravilno unesen broj!\n");
	 
	 while (broj >= 0) //Ako je broj pozitivan ili nula petlja će raditi
	 {
		 printf("Unesite %d broj: ",i++);
		 scanf("%d", &broj);
		 if (broj >= 0)
			 suma = suma + broj;
		 else
			 printf("Suma unesenih nenegativnih brojeva je %d\n",suma);
	 }

	system("pause");
}

/*Napisati program na programskom jeziku C u kome se učitavaju brojevi sve dok se ne unese negativan broj, a zatim na ekranu ispisati sumu unesenih brojeva djeljivih sa 5 ili 3*/

#include <stdio.h>
 main()
{
	 int  broj, suma=0,i;

	 printf("Unesite 1. broj: ");
	 scanf("%d", &broj);
	 
	 if (broj >= 0 && (broj % 3 == 0 || broj % 5 == 0))
		 suma += broj;

	 i = 2;
	 while (broj>=0)
	 {
		 printf("Unesite %d. broj: ",i);
		 scanf("%d", &broj);

		 if (broj >= 0 && (broj % 3 == 0 || broj % 5 == 0))
			 suma += broj;

		 i++;
	}

	 printf("Suma pravilno unesenih brojeva je: %d\n",suma);

	system("pause");
}
