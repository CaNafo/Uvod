Do While petlja 2

/*Napisati program na C-u u koji se unose brojevi sve dok se ne unese nula, a zatim na ekranu ispisati njihovu sumu.*/

#include <stdio.h>
main()
{
	double broj, sum = 0;

	// petlja se izvršava barem jednom
	do
	{
		printf("Unesite broj: ");
		scanf("%lf", &broj);
		sum += broj;
	} while (broj != 0.0);

	printf("Suma = %.2lf", sum);

	system("pause");
}

/*Napisati program na C-u u koji na ekranu ispisuje faktorijel unesenog broja.*/

#include <stdio.h>
main()
{
	int broj, i=1, fakt=1;
	
	printf("Unesite neki broj: ");
	scanf("%d", &broj);

	do {
		fakt *= i;
		i++;
	} while (i <= broj);

	printf("Faktorijel unesenog broja je %d.\n",fakt);
	system("pause");
}
