/*Napisati program na C-u u koji se unose pozitivni brojevi, ukoliko se unese negativan broj ili nula potrebno je na ekranu ispisati broj unesenih parnih pozitivnih brojeva.*/

#include <stdio.h>
main()
{
	int broj, brojParnih = 0 ,i=1;

	do {
		printf("Unesite %d. broj: ", i++);
		scanf("%d", &broj);

		if (broj > 0 && broj % 2 == 0)
			brojParnih++;
		
		} while (broj>0);

	printf("Parnih pozitivnih brojeva ima %d.\n",brojParnih);

	system("pause");
}

/*Napisati program na C-u u koji se unose dvocifreni brojevi sve dok je zbir njihovih cifara paran, na ekranu ispisati aritmetiƒçku sredinu ispravno unesenih brojeva.*/

#include <stdio.h>
main()
{
	int desetica, jedinica, brojac=0, broj, zbirJedDes;
	float ars = 0;

	do {
		printf("Unesite %d. broj: ", brojac+1);
		scanf("%d", &broj);

		if (broj > 9 && broj < 100)
		{
			desetica = broj / 10;
			jedinica = broj % 10;
			zbirJedDes = desetica + jedinica;
			if (zbirJedDes % 2 == 0)
			{
				ars += broj;
				brojac++;
			}
		}

	} while ((zbirJedDes % 2 == 0) && (broj > 9 && broj < 100));
	
	if (brojac > 0)
	{
		ars = ars / brojac;
		printf("Aritmeticka sredina pravilno unesenih brojeva je %.2f \n", ars);
	}
	else
		printf("Nema ni jedan pravilno unesen broj.\n");

	system("pause");
}
