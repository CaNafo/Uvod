/*Napisati program u programskom jeziku C kojim se omogućava unos jednog malog slova alfabeta, a zatim na ekran ispisati slovo koje je unijeto i njegovo veliko slovo. */

#include<stdio.h>
 main()
 {
	 char slovo, velikoSlovo;
	
	 printf("Unesite jedno malo slovo alfabeta: ");
	 scanf("%c", &slovo);

	 velikoSlovo ='A'+ slovo - 'a';

	 printf("Slovo koje ste unijeli je \"%c\", a njegovo veliko slovo je \"%c\".\n", slovo, velikoSlovo);
	 
	 system("pause");
}

/*Napisati program u programskom jeziku C kojim se omogućava unos dva slova alfabeta, a zatim se vrši njihova zamjena. Na ekran ispisati slova prije i poslije zamjene. (Zamjena ne smije biti prividna već se mijenja sadržaj varijabli) */

#include<stdio.h>
 main()
 {
	 char prvoSlovo, drugoSlovo, pom;
	
	 printf("Unesite dva slova alfabeta: ");
	 scanf("%c %c", &prvoSlovo, &drugoSlovo);
	 
	 printf("\nSlova prije zamjene su: %c %c.\n",prvoSlovo, drugoSlovo);

	 pom = prvoSlovo;
	 prvoSlovo = drugoSlovo;
	 drugoSlovo = pom;

	 printf("\nSlova nakon zamjene su: %c %c.\n", prvoSlovo, drugoSlovo);
	 
	 system("pause");
}
Napomena!!!
Ukoliko na ekranu želite ispisati neki od rezervisanih karaktera kao: " % \ . Neophodno je to napisati u sledećem formatu: printf("\\ %% \" "); .
