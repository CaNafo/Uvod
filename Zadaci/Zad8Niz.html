/*Izračunati dužinu niza karaktera koji se učitava sa tastature koristeći funkciju strlen.*/

#include <stdio.h>

main()
{

	char niz[20];
	printf("Unesite niz: ");
	scanf("%s", &niz);
	printf("\nDužine niza je: %d\n", strlen(niz));

	system("PAUSE");

}

/*Učitati niz karaktera sa tastature pomoću funkcije gets, a zatim izračunati dužinu unesenog stringa.*/

#include <stdio.h>

main()
{

	char niz[20];
	printf("Unesite niz: ");
	gets(niz);
	printf("\nDužine niza je: %d\n", strlen(niz));

	system("PAUSE");

}

/*Učitati niz karaktera sa tastature pomoću funkcije gets, a zatim napraviti novi niz koji će biti identičan unesenom nizu, ali će na mjestima unesnih razmaka da ima karakter 'R'.*/

#include <stdio.h>

main()
{

	int i;

	char niz[20], novi[20];
	printf("Unesite niz: ");

	gets(niz);

	for (i = 0; niz[i] != '\0'; i++)

		if (niz[i] == ' ') //isspace(niz[i])

			novi[i] = 'R';
		else

			novi[i] = niz[i];

	novi[i] = '\0';
	printf("Novi niz je: %s\n", novi);

	system("PAUSE");

}
