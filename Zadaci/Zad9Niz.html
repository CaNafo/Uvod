/*Dat je niz od maksimalno 15 celobrojnih elemenata. Napravite novi niz koji ce sadrzati elemente koji su veci od aritmeticke sredine prvog niza.*/

#include <stdio.h>
#define MAX 15


int main()
{
    int nizA[MAX], nizB[MAX];
    int n, i, j = 0, arSr = 0;

    //Unosenje velicine niza
     do{
        printf("Unesite velicinu niza(max 15): ");
        scanf("%d", &n);
     }while(n < 1 || n > 15);

     //Popunjavanje niza
     for(i = 0; i < n; i++){
        printf("Unesite %d. element niza: ", i+1);
        scanf("%d", &nizA[i]);
        arSr += nizA[i];     //suma niza
     }

     arSr = arSr / n;       //racunanje aritmeticke sredine

     //popunjavanje drugog niza
     for(i = 0; i < n; i++){
        if(nizA[i] > arSr){
            nizB[j] = nizA[i];
            j++;
        }
     }

     //ispis novog niza
     printf("\nAritmeticka sredina niza je: %d \nNizB = [ ", arSr);
     for(i = 0; i < j; i++){
        if(i == j - 1){
            printf("%d].", nizB[i]);
        }else {
            printf("%d, ", nizB[i]);
        }
     }

    system("pause");
}

/*Napraviti i ispisati Fibonaccijev niz za N elemenata, za N vece od 3.*/

#include <stdio.h>


int main()
{
    int n, i;

    //Unosenje velicine niza
     do{
        printf("Unesite velicinu niza(mora biti vece od 3): ");
        scanf("%d", &n);
     }while(n < 3);

     int fibonacciNiz[n];

     //Popunjavanje niza
     fibonacciNiz[0] = 1;
     fibonacciNiz[1] = 1;
     for(i = 2; i < n; i++){
        fibonacciNiz[i] = fibonacciNiz[i-1] + fibonacciNiz[i-2];
     }

     //ispis niza
     printf("\nFibonaccijev niz = [ ");
     for(i = 0; i < n; i++){
        if(i == n - 1){
            printf("%d].\n", fibonacciNiz[i]);
        }else {
            printf("%d, ", fibonacciNiz[i]);
        }
     }

    system("pause");
}
