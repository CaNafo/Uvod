/*Napisati program u C-u kojim se omogućava unos jednog cijelog broja, jednog realnog broja, zaokruženog na dvije decimale, i jednog karaktera, zatim ih ispisati na ekran respektivno*/

 

#include<stdio.h>
 main()
{
	int CijeliBroj;
	float RealanBroj;
	char Karakter;

	printf("Unesite cio broj, realan broj i karakter respektivno: ");

	scanf("%d %f %c", &CijeliBroj, &RealanBroj, &Karakter);

	printf("Unešeni cijeli broj je: %d\nUnešeni realan broj je: %.2f\n", CijeliBroj, RealanBroj);
	
	printf("Unešeni karakter je: %c\n", Karakter);

	system("pause");
}
