/*Napisati program koji učitava matricu, a potom je ispisuje.*/

#include <stdio.h>

int main()

{

	int i, j;

	int Matrica[3][3];

	printf("Unesite elemente matrice: \n");
	for (i = 0; i < 3; i++) {

		for (j = 0; j < 3; j++) {

			scanf("%d", &Matrica[i][j]);
		}

	}

	printf("\nIspis matrice:\n");

	for (i = 0; i < 3; i++)
	{

		for (j = 0; j < 3; j++)

		{
			printf("%d ", Matrica[i][j]);

		}

		printf("\n");
	}

	system("PAUSE");

}

/*Napisati program koji učitava matricu, a potom sabira elemente koji nisu na glavnoj dijagonali matrice.*/

#include <stdio.h>

int main()

{

	int i, j, suma = 0;

	int Matrica[3][3];

	printf("Unesite elemente matrice: \n");

	for (i = 0; i < 3; i++) {
		for (j = 0; j < 3; j++) {

			scanf("%d", &Matrica[i][j]);

		}
	}

	printf("\nIspis matrice:\n");

	for (i = 0; i < 3; i++)
	{

		for (j = 0; j < 3; j++)

		{

			printf("%d ", Matrica[i][j]);
		}

		printf("\n");

	}
	for (i = 0; i < 3; i++)

	{

		for (j = 0; j < 3; j++)
		{

			if (i != j)

				suma += Matrica[i][j];

		}
	}

	printf("\nSuma elemenata koji nisu na glavnoj dijagonali je: %d \n\n", suma);

	system("pause");

}
