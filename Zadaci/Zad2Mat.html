/*Sa tastature unijeti dimenziju i elemente cjelobrojnog niza, a na ekran prikazati histogram vrijednosti elemenata unetog niza u vidu vertikalnih stubica kao na slici:*/

#include<stdio.h>
main()
{
	int i, n, niz[20], max = 0, red, brojac;
	do
	{
		printf("Unesi duzinu niza: ");
		scanf("%d", &n);
	} while (n < 1 || n>20);
	printf("Unesi %d elemenata niza: \n", n);
	for (i = 0; i < n; i++)
		scanf("%d", &niz[i]);
	max = niz[0];
	for (i = 1; i < n; i++)
		if (niz[i]>max)
			max = niz[i];
	printf("\n");
	for (red = 1; red <= max; red++)
	{
		for (brojac = 0; brojac < n; brojac++)
			if (niz[brojac]>max - red)
				printf("* ");
			else
				printf("  ");
		printf("\n");
	}
	system("pause");
}
