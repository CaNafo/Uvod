/*Korištenjem pokazivača, napisati funkciju koja određuje parnost unesenog broja, a prilikom unosa broja 0 prekida se rad programa.*/

#include <stdio.h>

void parnost(int *pbr) {

	if (*pbr % 2 == 0)
		printf("\nBroj je paran.");
	else
		printf("\nBroj je neparan.");

}

void main() {
	int br, *pbr;

	pbr = &br;

	printf("\nUpisite broj: ");
	scanf_s("%d", pbr);

	while (*pbr != 0) {
		parnost(pbr);
		printf("\nUpisite broj: ");
		scanf_s("%d", pbr);
	}


	system("pause");
}

/*Upisati 5 brojeva u niz. U funkciji,korištenjem pokazivača, naći njihovu aritmetičku sredinu i ispisati je u glavnom programu.*/

#include <stdio.h>

float arit(int *pok) {
	int i, suma = 0;
	float ar;

	for (i = 0; i < 5; i++)
		suma += *(pok + i);

	ar = suma / 5.0;

	return ar;
}

void main() {
	int i, niz[5], *pok;
	float ars;

	pok = &niz[0];

	for (i = 0; i < 5; i++) {
		printf("\Unesite %d. broj: ", i + 1);
		scanf("%d", pok + i);
	}

	ars = arit(pok);

	printf("\nAritmeticka sredina je %.2f\n", ars);

	system("pause");
}
