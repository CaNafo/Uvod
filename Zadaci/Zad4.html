/*Napisati program u C-u kojim se omogućava unos sekundi, a na ekranu se ispisuje koliko je to sati minuta i sekundi u fomatu hh:mm:ss*/

#include <stdio.h>

 main()
{
	int unosSec, sati, min, sec;

	printf("\nUnesite broj sekundi: ");
	scanf("%d", &unosSec);

	sati = unosSec / 3600;
	min = (unosSec-(sati*3600)) / 60;
	sec = (unosSec-(sati*3600)) % 60;
	
        printf("\n%d s = %d h : %d min : %d sec", unosSec, sati, min, sec);

        system("pause");
}

/*Napisati program u C-u kojim se omogućava unos trocifrenog broja, a na ekranu se ispisuju njegove cifre i njihov proizvod*/

#include <stdio.h>

 main()
{
	int broj, sto, des, jed, proizvod;
	

	printf("\nUnesite trocifreni broj: ");
	scanf("%d", &broj);

	sto = broj / 100;
	des = (broj % 100) / 10;
	jed = (broj % 100) % 10;
	proizvod = sto * des * jed;

	printf("\nPrva cifra: %d", sto);
	printf("\nDruga cifra: %d", des);
	printf("\nTreća cifra: %d", jed);
	printf("\nProizvod je: %d", proizvod); 


        system("pause");
}
