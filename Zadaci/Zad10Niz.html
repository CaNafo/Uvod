/*Dat je niz od maksimalno 15 celobrojnih elemenata. Učitati n elemenata a zatim naći 2 elementa čija je razlika najveća. Sprovesti zaštitu unosa nad podatkom n.*/

#include <stdio.h>
#define MAX 15

//Ako zelite da ispisete indekse elemenata izmedju kojih je najveca razlika, sklonite komentare koji sadrze min i max Indeks.

int main()
{
    int niz[MAX];
    int maxEl, minEl, n, i, razlika;
    //int maxIndeks = 0, minIndeks = 0;

    //Unosenje velicine niza
     do{
        printf("Unesite velicinu niza(max 15): ");
        scanf("%d", &n);
     }while(n < 1 || n > 15);

     //Popunjavanje niza
     for(i = 0; i < n; i++){
        printf("Unesite %d. element niza: ", i+1);
        scanf("%d", &niz[i]);
     }

     //Nalazenje najmanjeg i najveceg elementa niza
     maxEl = niz[0];
     minEl = niz[0];
     for(i = 0; i < n; i++){
        if(niz[i] > maxEl){
            maxEl = niz[i];
            //maxIndeks = i;
        }

        if(niz[i] < minEl){
            minEl = niz[i];
            //minIndeks = i;
        }
     }
     razlika = maxEl - minEl;

     printf("\nNajveca razlika izmedju elemenata je %d.\n", maxEl - minEl);

     //printf("\nNajveca razlika je izmedju elemenata niz[%d] = %d i niz[%d] = %d, a razlika je %d.", maxIndeks, maxEl, minIndeks, minEl, razlika);

    system("pause");
}
