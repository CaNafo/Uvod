/*Napisati program u programskom jeziku C za unos dva broja, a zatim na ekran ispisati broj koji je veći*/

#include <stdio.h>

main()
 {
     int a, b,max;
     printf("Unesite dva broja: ");
     scanf("%d %d", &a, &b);
     if(a>b)
     {
       max=a;
             }
          else 
          {
            max=b;
                  }
      printf("Veci broj je: %d",max);
     
      system("pause");
}

/*Napisati program u programskom jeziku C u kome se unosi broj, a zatim na ekranu ispisati apsolutnu vrijednost tog broja */

#include <stdio.h>
main()
 {
     int broj;
     printf("Unesite broj: ");
     scanf("%d", &broj);
     if(broj<0)
     {
       broj=broj*(-1);
             }
          
      printf("Apsolutna vrijednost unešenog broja je: |%d|",broj);
     
      system("pause");
}
Napomena!!!
Ispod komandi "if" i "else" nije neophodno, ali je poželjno pisati vitičaste zagrade {}, ukoliko se one ne napišu kompajler će smatrati da samo jedna linija koda ispod ovih komandi spada u uslov. Sve ostale linije koda se smatraju normalnim tokom programa i biće izvršene bez obzira na uslov.
