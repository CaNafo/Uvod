/*Napisati program na programskom jeziku C koji za unijeti broj 'n' ispisuje "piramidu" slova npr. za n=5
         A
       BCD
      EFGHI
   JKLMNOP
QRSTUVWXY
(U koliko se dođe do posljednjeg slova alfabeta potrebno je nastaviti ispisivati slova od početka npr. X Y Z A B...)*/

#include<stdio.h>
main()
{
	int i, j, n;
	char slovo = 'A'; 

	printf("Unesite broj n: ");
	scanf("%d", &n);

	for (i = 0; i < n; i++) 
	{
		for (j = 0; j< n-i; j++) 
		{						
			printf(" ");			
		}			   				
		for (j = 0; j<= i*2; j++)
		{				
			if(slovo>'Z') //Uslov za restartovanje brojača slova
				slovo = 'A';

			printf("%c", slovo++);			
		}
	
		printf("\n");
	}

	system("pause");
}

/*Napisati program na programskom jeziku C koji za unijeti broj 'n' ispisuje "piramidu" slova npr. za n=5
         A
       BcD
      EfGhI
   JkLmNoP
QrStUvWxY
(U koliko se dođe do posljednjeg slova alfabeta potrebno je nastaviti ispisivati slova od početka npr. X Y Z A B...)*/

#include<stdio.h>
main()
{
	int i, j, n;
	char slovo = 'A'; 

	printf("Unesite broj n: ");
	scanf("%d", &n);

	for (i = 0; i < n; i++) 
	{
		for (j = 0; j< n-i; j++) 
		{						
			printf(" ");			
		}			   				
		for (j = 0; j<= i*2; j++, slovo++)
		{				
			if(slovo>'Z') //Uslov za restartovanje brojača slova
				slovo = 'A';
			
			if(j%2==0)  //Uslov koji provjerava da li se slovo nalazi na parnom ili
				printf("%c", slovo);  //neparnom mjestu u koloni
			else
				printf("%c", slovo-'A'+'a');//Pošto su slova dekadni bojevi u ASCII tabeli
								//dozvoljene su operacije oduzimanja i sabiranja
					   //tako da lako dobijamo malo slovo od velikog i obrnuto
		}
	
		printf("\n");
	}

	system("pause");
}
