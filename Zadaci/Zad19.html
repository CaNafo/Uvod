/*Napisati program na programskom jeziku C u kome se unose brojevi dok su negativni. Za svaki učitani broj izračunati i ispisati njegovu apsolutnu vrijednost!*/

#include <stdio.h>
main()
{
	int broj;

	printf("\Unesite broj broj: ");
	scanf("%d", &broj);

	while (broj<0)
	{
		printf("\nApsolutna vrijednost %d je %d\n\n", broj, broj*(-1));

		printf("Unesite broj: ");
		scanf("%d", &broj);
	}

	system("pause");
}

/*Napisati program na programskom jeziku C koji računa sumu brojeva djeljivih sa 3 od 0 do N.*/

#include <stdio.h>
main()
{
	int n, i=0, suma=0;

	printf("\Unesite gornju granicu N: ");
	scanf("%d", &n);

	while (i<n) //Ovaj uslov ne uračunava gornju granicu
	{			//Ako treba uračunati i gornju granicu tada bi pisali i<=n
		if (i % 3 == 0)
			suma = suma + i;
		i++;
	}

	printf("Suma brojeva od 0 do %d je: %d ", n, suma);

	system("pause");
}
