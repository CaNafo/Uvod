/*Napisati program u programskom jeziku C gdje korisnik unosi dva broja, a zatim se na ekranu ispisuje rezultat oduzimanja manjeg broja od veceg*/

#include<stdio.h>
main()
{
	float a, b;
	printf("Unesite dva broja: ");
	scanf("%f %f", &a, &b);

	if (a > b)
		printf("Rezultat oduzimanja broja %.2f od %.2f je %.2f\n", b, a, a - b);
	else
		printf("Rezultat oduzimanja broja %.2f od %.2f je %.2f\n", a, b, b - a);

	system("pause");
}

/*Napisati program u programskom jeziku C gdje korisnik unese slovo, a zatim, u koliko je slovo malo, na ekranu se ispisuje njegovo veliko slovo i obrnuto. Takođe ispisati odgovarajuću poruku u koliko je unos pogrešan.*/

#include<stdio.h>
main()
{
	char slovo;
	printf("Unesite neko slovo: ");
	scanf("%c", &slovo);

	if (slovo >= 'a' && slovo <= 'z')
	{
		slovo = slovo - 'a';
		slovo = 'A' + slovo;
		printf("Veliko slovo je: %c\n", slovo);
	}
	else if (slovo >= 'A' && slovo <= 'Z')
	{
		slovo = slovo - 'A';
		slovo = 'a' + slovo;
		printf("Malo slovo je: %c\n", slovo);
	}
	else
		printf("Pogresan unos!\n");
	system("pause");
}
