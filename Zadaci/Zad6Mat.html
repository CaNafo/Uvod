/*Napisati program u programskom jeziku C za unos matrice dimenzija M N, a zatim tu matricu transformisati tako da se svaka parna kolona rotira u lijevu stranu onoliko puta kolika je brojna vrijednost tekućeg reda. Npr. ako su elementi drugog reda : 1 2 3 4 nakon transformacije treba da izgleda kao 3 4 1 2. (Redovi se broje od 0) Ispisati matricu prije i nakon transformacije*/

#include<stdio.h>
#include<stdlib.h>

int main()
{
	int mat[100][100], m, n, j, i, k, pom;

	printf("Unesite dimenzije matrice M N: ");
	scanf_s("%d %d", &m, &n);

	for (i = 0; i < m; i++)
	{
		for (j = 0; j < n; j++) {
			printf("Unesite %d. element %d. reda: ", j+1, i+1);
			scanf_s("%d", &mat[i][j]);
		}
	}
	
	printf("Matrica prije transformacije:\n\n");
	for (i = 0; i < m; i++)
	{
		for (j = 0; j < n; j++) {
			printf("%d ", mat[i][j]);
		}
		printf("\n");
	}
	printf("\n");
	for (i = 0; i < m; i++)
	{
			if (i % 2 == 0) {
				for (k = 0; k < i; k++) {
					pom = mat[i][n-1];
					for (j = n-1; j > 0;j--)
						mat[i][j] = mat[i][j-1];
					mat[i][0] = pom;
				}
			}
	}

	printf("Matrica poslije transformacije:\n\n");
	for (i = 0; i < m; i++)
	{
		for (j = 0; j < n; j++) {
			printf("%d ", mat[i][j]);
		}
		printf("\n");
	}
	printf("\n");

	system("pause");

}

/*Napisati program za unos kvadratne matrice dimenzije N, zatim napraviti transformisanu matricu bez korištenja pomoćne matrice. Ispisati matricu prije i nakon transformacije*/

#include<stdio.h>
#include<stdlib.h>

int main()
{
	int mat[100][100], m, j, i, pom;

	printf("Unesite dimenziju kvadratne matrice: ");
	scanf_s("%d", &m);

	for (i = 0; i < m; i++)
	{
		for (j = 0; j < m; j++) {
			printf("Unesite %d. element %d. reda: ", j+1, i+1);
			scanf_s("%d", &mat[i][j]);
		}
	}
	
	printf("\n\nMatrica prije transformacije:\n\n");
	for (i = 0; i < m; i++)
	{
		for (j = 0; j < m; j++) {
			printf("%d ", mat[i][j]);
		}
		printf("\n");
	}
	
	for (i = 0; i < m; i++)
	{
		for (j = i+1; j < m; j++) {
			pom = mat[i][j];
			mat[i][j] = mat[j][i];
			mat[j][i] = pom;
		}
		printf("\n");
	}

	printf("Matrica poslije transformacije:\n\n");
	for (i = 0; i < m; i++)
	{
		for (j = 0; j < m; j++) {
			printf("%d ", mat[i][j]);
		}
		printf("\n");
	}
	system("pause");

}
